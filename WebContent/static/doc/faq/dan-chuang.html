弹窗

弹窗

1. 介绍

基于PKUI的desktop，默认以抽屉的方式从右侧滑出一个弹窗，可应用于一般性的功能模块的增改查。

也就是说，它以覆盖某个层的方式展现的。（得有一个现成的容器）

对于列表页来说，上述方式是可行的，

但是，对于那些相对独立的功能来说，这种滑出式的方式就不可行了，比如：修改密码功能。

这种情况，你可以使用第三方插件 layer（已经引入）。

2. 使用

参考：

  • desktop/js/page/toolbarUserDropdown.js
  • desktop/tpl/system/user/modifyPassword.html

示例：

// desktop/js/page/toolbarUserDropdown.js
<button type="button" id="modifyPassword" class="btn btn-warning-2" onclick="modifyPassword()">改密码</button>
<script>
  function modifyPassword() {
    var 
      viewUrl = "__CTX__/static/desktop/tpl/system/user/modifyPassword.html",
      modelUrl = "__CTX__/admin/getCurrentSysUser",
      $layerContent,
      layerIndex
    ; 
    seajs.use( "template", function ( Template ) {

      layerIndex = layer.open( {
          id: "modifySysUserPassword",
          title: "修改用户密码",
          type: 1,
          skin: 'layui-layer-rim', //加上边框
          area: ['600px', '420px'], //宽高
          content: '',
          success: function ( layero ) {

              $layerContent = layero.find( ".layui-layer-content" );

              // 开启loading
              $layerContent.isLoading();

              Template.getModelAndView( viewUrl, modelUrl, function ( htmlString ) {
                  if ( ! $layerContent ) {
                      return;
                  }
                  // 关闭 loading
                  $layerContent.isLoading( "hide" );

                  $layerContent.html( htmlString );
              } );

          },
          end: function () {
              $layerContent = null;
          }
      } );
      // 密码修改成功后的事件通知
      $( document ).one( "modifySysUserPasswordSuccess", function () {
          layer.close( layerIndex );
          window.location = window.PKUI.ctxPath;
      } );

    } );
  }
</script>

// desktop/tpl/system/user/modifyPassword.html
<form class="da-form"
    action="{% system.user.modifyPassword.updateSysUserPassword %}" method="post"
    data-pkui-component="validator|form"
    data-done-callback="updateSysUserPasswordDoneCallback" >


    <div class="da-form-heading"><!--修改用户密码--></div>
    <div class="da-form-body">
        <table class="da-form-table da-table da-table-striped">
            ......
        </table>
    </div>
    <div class="da-form-footer">
        <button type="submit" class="btn btn-success-2">确定</button>
    </div>
</form>

<script>
    // 请求发送成功，对服务器端返回的数据进行处理
    function updateSysUserPasswordDoneCallback($form, jsonResult) {
        console.info( jsonResult );
        jsonResult = window.PKUI.handleJsonResult( jsonResult );
        console.info( jsonResult );
        // 服务器端处理成功
        if ( jsonResult.success ) {
            // 提示
            layer.alert( '修改密码成功！', { icon: 1 }, function ( index ) {
                layer.close( index );
                // 生成一个事件
                $( document ).trigger( "modifySysUserPasswordSuccess" );

            } );
        }
        // 服务器端处理失败
        else {
            // 提示
            layer.alert( '修改密码失败！', { icon: 2 } );
        }
    }
</script>
