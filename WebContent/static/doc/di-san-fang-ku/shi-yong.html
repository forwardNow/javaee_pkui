使用

使用（第三方库）

1. 传统方式

通过<script>标签引入JS文件，并通过手动调整<script ...></script> 的顺序 来处理依赖关系。

// 依赖（在 第三方库 之前执行）
<script src="${ctx}/static/pkui/lib/jquery/1.11.3.x/jquery.js"></script>
// 第三方库
<script src="${ctx}/static/pkui/lib/component/datetimepicker/2.5.4.x/datetimepicker.js"></script>

// 使用
<input type="text" id="myDatetimepicker">
<script>
    jQuery( document ).ready( function ( $ ) {
        $( "#myDatetimepicker" ).datetimepicker( {
            // 参数
        } );
    } );
</script>

优点：

  • 无兼容性问题
  • 直观

缺点：

  • 得手动处理依赖关系
  • 当引入大量JS文件时，使页面变得凌乱，且不便于维护

2. 模块加载器方式（使用seajs）

模块加载器让JavaScript可以以“包”（package）的方式来加载并使用JS。

2.1 使用绝对路径引入

<input type="text" id="myDatetimepicker">
<script>
    seajs.use( [
        window.PKUI.ctxPath + "/static/pkui/lib/jquery/1.11.3.x/jquery", // 不加“.js”后缀名   
        window.PKUI.ctxPath + "/static/pkui/lib/component/datetimepicker/2.5.4.x/datetimepicker" // 不加“.js”后缀名
    ], function ( $ ) {
        $( "#myDatetimepicker" ).datetimepicker( {
            // 参数
        } );
    } );
</script>

2.2 使用模块别名引入

模块别名，参阅 模块

<input type="text" id="myDatetimepicker">
<script>
    seajs.use( [
        "jquery",    
        "datetimepicker"
    ], function ( $ ) {
        $( "#myDatetimepicker" ).datetimepicker( {
            // 参数
        } );
    } );
</script>

2.3 比较

对已经预定义好的第三方模块，推荐使用模块别名，因为：

  • 简短
  • 模块别名对应的JS文件是已经修改过的（修改为CMD模块，并处理依赖关系）

3. 引入自定义的第三方组件/插件

当PKUI里预定义的组件/插件 无法满足你的要求时：

  • 情况1：在网上找到了解决你问题的第三方组件/插件
  • 情况2：使用驱动某外设（虹膜识别）的JS插件

也就是说，你要引入的JS，有这几种形式：

  • jQuery插件
    • 无依赖
    • 有依赖
  • 非jQuery插件，将使用的接口暴露在全局作用域（如富文本编辑器 window.UM）
    • 无依赖
    • 有依赖

3.1 引入 jQuery插件

由于jQuery为预加载的模块，所以只要 pkui/pkui.js 被正确引入了，

对于在 pkui/pkui.js 之后引入的模块，jQuery是可以直接使用的。

3.1.1 无依赖

直接通过 seajs.use() 使用：

proj
|-- static
    |-- pkui
    |-- js
        |-- jquery.datepicker2.js

<input type="text" id="addDate">
<script>
    seajs.use( [ 
        "jquery",  
        window.PKUI.ctxPath + "/static/js/jquery.datepicker2"
    ], function ( $ ) {
        $( "#addDate" ).datepicker2( { ... } );
    } );
</script>

3.1.2 有依赖

这时得修改源文件，将其改造为CMD模块以定义其依赖关系，修改方法如下：

// 原文件
;+function ( $, undefined ) {
    ...
}( jQuery );

// 修改后
define( function ( require ) {
    // 依赖
    require( "jquery" );

    // 原文件
    ;+function ( $, undefined ) {
        ...
    }( jQuery );
} );

通常是有配套的国际化文件，而且 国际化文件 得在插件之后执行，也就是说 国际化文件 依赖 插件；

有时，还要在JS中引入样式文件。

示例1：将依赖链的末端模块作为组件的入口

zh-cn.js 依赖 jquery.datepicker2.js 依赖 jquery

zh-cn.js 不被任何JS依赖，其作为依赖链的末端模块，对其进行修改。

由于 seajs 无法在外部定义依赖关系，因此只能有JS文件全部改造为CMD模块，然后定义其依赖关系。

proj
|-- static
    |-- pkui
    |-- js
        |-- jquery.datepicker2.js
        |
        |-- lang
        |   |-- zh-cn.js
        |
        |-- datepicker2.css

/* zh-cn.js */
define( function ( require ) {
    // 引入 jquery
    require( "jquery" );
    // 使用相对路径引入 jquery.datepicker2.js （去掉“.js”后缀）
    require( "../jquery.datepicker2" );
    // 引入样式文件
    require( "../datepicker2.css" );

    // 原文件
    ;+function ( $, undefined ) {
        ...
    }( jQuery );
} );

<input type="text" id="addDate">
<script>
    seajs.use( [ 
        "jquery",  
        window.PKUI.ctxPath + "/static/js/lang/zh-cn"
    ], function ( $ ) {
        $( "#addDate" ).datepicker2( { ... } );
    } );
</script>

示例2：进行封装

新建一个 wrapper.js 文件，引入零碎的文件。

proj
|-- static
    |-- pkui
    |-- js
        |-- jquery.datepicker2.js
        |
        |-- lang
        |   |-- zh-cn.js
        |
        |-- datepicker2.css
        |
        |-- wrapper.js

/* jquery.datepicker2.js */
define( function ( require ) {
    // 引入 jquery
    require( "jquery" );

    // 原文件
    ;+function ( $, undefined ) {
        ...
    }( jQuery );
} );


/* zh-cn.js */
define( function ( require ) {
    // 引入 jquery
    require( "jquery" );
    // 使用相对路径引入 jquery.datepicker2.js （去掉“.js”后缀）
    require( "../jquery.datepicker2" );

    // 原文件
    ;+function ( $, undefined ) {
        ...
    }( jQuery );
} );

/* wrapper.js */
define( function ( require ) {
    // 引入 jquery
    require( "jquery" );
    // 引入样式文件
    require( "./datepicker2.css" );

    // 引入 datepicker2
    require( "./lang/zh-cn" );
} );

<input type="text" id="addDate">
<script>
    seajs.use( [ 
        "jquery",  
        window.PKUI.ctxPath + "/static/js/wrapper"
    ], function ( $ ) {
        $( "#addDate" ).datepicker2( { ... } );
    } );
</script>

3.2 引入非jQuery插件

对于非jQuery插件，需要注意：

  • 不要污染全局作用域
  • 不要覆盖 全局变量$

3.2.1 无依赖

传统的插件：

proj
|-- static
    |-- hmsb    // 虹膜识别
        |-- index.js

---------

/* 原始 index.js */

function $( id ) {
    return document.getElementById( id );
}
function hmsb( arg1, arg2, arg3 ) {
    ...
}

---------

// 使用
<script src="${ctx}/static/hmsb/index.js"></script>
<button type="button" onclick="doHmsb()">识别</button>
<script>
    function doHmsb() {
        hmsb( ... );
    }
</script>

缺点：

  • 占用全局变量 $
  • 未对其进行封装，而在全局作用域暴露过多东西

现在，将其拷贝一份并重命名为 wrapper.js，然后改造为CMD模块并使用：

proj
|-- static
    |-- hmsb    // 虹膜识别
        |-- index.js
        |-- wrapper.js

---------

/* wrapper.js */
define( function () {

    // 原文件内容
    function $( id ) {
        return document.getElementById( id );
    }
    function hmsb( arg1, arg2, arg3 ) {
        ...
    }

    // 暴露出去的API
    return hmsb;
} );

---------

// 使用
<button type="button" id="doHmsbBtn">识别</button>
<script>
    seajs.use( [ window.PKUI.ctxPath + "/static/hmsb/wrapper" ], function ( hmsb ) {
        jQuery( "#doHmsbBtn" ).on( "click", function () {
            hmsb( ... );
        } );  
    } );
</script>

优点：

  • 不污染全局作用域

3.2.2 有依赖

原始：

proj
|-- static
    |-- hmsb    // 虹膜识别
        |-- index.js
        |-- util.js
        |-- style.css

---------

/* util.js */
function $( id ) {
    return document.getElementById( id );
}

---------

/* index.js */
function hmsb( arg1, arg2, arg3 ) {
    var value = $( arg1 ).value;
    ...
}

---------

// 使用
<link rel="stylesheet" href="${ctx}/static/hmsb/style.css">
<script src="${ctx}/static/hmsb/util.js"></script>
<script src="${ctx}/static/hmsb/index.js"></script>
<button type="button" onclick="doHmsb()">识别</button>
<script>
    function doHmsb() {
        hmsb( ... );
    }
</script>

改造：

proj
|-- static
    |-- hmsb    // 虹膜识别
        |-- index.js
        |-- util.js
        |-- style.css
        |
        |-- util.wrapper.js
        |-- index.wrapper.js


---------

/* util.wrapper.js */
define( function () {

    // 原始内容
    function $( id ) {
        return document.getElementById( id );
    }

    // 暴露接口
    return {
        $: $
    };
} );


---------

/* index.wrapper.js */
define( function ( require ) {
    // 引入样式文件
    require( "./style.css" );

    // 引入依赖
    var Util = require( "./util.wrapper" );

    // 重新设置原始内容中使用到的工具函数    
    var $ = Util.$;

    // 原始内容
    function hmsb( arg1, arg2, arg3 ) {
        var value = $( arg1 ).value;
        ...
    }

    // 暴露接口
    return hmsb;
} );

---------

// 使用
<button type="button" id="doHmsbBtn">识别</button>
<script>
    seajs.use( [ window.PKUI.ctxPath + "/static/hmsb/index.wrapper" ], function ( hmsb ) {
        jQuery( "#doHmsbBtn" ).on( "click", function () {
            hmsb( "doHmsbBtn" );
        } );  
    } );
</script>
